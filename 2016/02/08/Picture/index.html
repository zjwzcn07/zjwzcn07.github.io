USACO-5.5-Picture

题意

N个矩形贴在二维平面上，有可能有重合覆盖，求所有矩阵组成的集合的周长
数据为N行，每行为矩阵的左下角和右上角坐标

解

每条横边从下到上排列，纵边从左到右排列
每次遇到起始边则该点++，否则--
当该点标记从0变1时，ans++
从1变0时，ans++

代码

struct node
{
    int p,s,e;
    int start;
}H[10000+1],V[10000+1];
int HH[20000+3],VV[20000+3];
bool cmp(node a,node b)
{
    if(a.p==b.p)
    {
        return a.start>b.start;
    }
    else
        return a.p<b.p;;
}
int N,v,h;
int main()
{
    scanf("%d",&N);
    for(int i=0;i<N;i++)
    {
        int x1,y1,x2,y2;
        cin>>x1>>y1>>x2>>y2;
        V[v].p=x1;
        V[v].s=y1;
        V[v].e=y2;
        V[v++].start=1;
        V[v].p=x2;
        V[v].s=y1;
        V[v].e=y2;
        V[v++].start=0;
        H[h].p=y1;
        H[h].s=x1;
        H[h].e=x2;
        H[h++].start=1;
        H[h].p=y2;
        H[h].s=x1;
        H[h].e=x2;
        H[h++].start=0;
    }
    sort(H,H+h,cmp);
    sort(V,V+v,cmp);
    int ans=0;
    for(int i=0;i<h;i++)
    {
        if(H[i].start)
        {
            for(int j=H[i].s;j<H[i].e;j++)
            {
                HH[j+10000]++;
                if(HH[j+10000]==1)
                    ans++;
            }
        }
        else
        {
            for(int j=H[i].s;j<H[i].e;j++)
            {
                HH[j+10000]--;
                if(HH[j+10000]==0)
                    ans++;
            }
        }
        if(V[i].start)
        {
            for(int j=V[i].s;j<V[i].e;j++)
            {
                VV[j+10000]++;
                if(VV[j+10000]==1)
                    ans++;
            }
        }
        else
        {
            for(int j=V[i].s;j<V[i].e;j++)
            {
                VV[j+10000]--;
                if(VV[j+10000]==0)
                    ans++;
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
